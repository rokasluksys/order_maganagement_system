const orderSchema = new mongoose.Schema({
  user: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User'
  },
  date: Date,
  customer: String,
  orderDescription: String,
  designService: Boolean,
  format: String,
  singleSidedPrint: Boolean,
  doubleSidedPrint: Boolean,
  paper: String,
  lamination: String,
  binding: String,
  uv3DCoating: Boolean,
  otherPostPrintingWorks: String,
  quantity: Number,
  additionalOrderInformation: String,
  deliveryCost: Number,
  shippingDate: Date,
  orderPrice: Number,
  invoiceNumber: String,
  invoiceStatus: String,
  additionalNotes: String,
  customerContactDetails: String
})

data - date
uzsakovas - customer
uzsakymas - orderDescription
maketavimoPaslauga - designService
formatas - format
vienpuseSpauda - singleSidedPrint
dvipuseSpauda - doubleSidedPrint
popierius - paper
laminavimas - lamination
irisimas - binding
lakavimas3DUV - uv3DCoating
kitiPospaudiminiaiDarbai - otherPostPrintingWorks
tirazas - quantity
papildomaUzsakymoInformacija - additionalOrderInformation
pristatymoKaina - deliveryCost
issiuntimoData - shippingDate
uzsakymoKaina - orderPrice
saskaitosNumeris - invoiceNumber
saskaitosBusena - invoiceStatus
pastabos - additionalNotes
uzsakovoKontaktiniaiDuomenys - customerContactDetails
